class Solution {
    public int[] singleNum(int[] arr) {
        int ans[]= new int[2];
        int j=0;
        HashMap <Integer,Integer> hm = new HashMap<>();
        for(int i=0;i<arr.length;i++){
            hm.put(arr[i],hm.getOrDefault(arr[i],0)+1);
        }
        for(Map.Entry <Integer,Integer> e:hm.entrySet()){
            if(e.getValue()==1){
                ans[j]=e.getKey();
                j++;
            }
        }
        Arrays.sort(ans);
        return ans;
        
    }
}
