class Solution {
    public String longestCommonPrefix(String[] strs) {
        int n=strs[0].length();
        String prefix="";
        for(int i=1;i<=n;i++){
            prefix=strs[0].substring(0,i);
            for(int j=1;j<strs.length;j++){
                if(!(strs[j].startsWith(prefix))){
                    return strs[0].substring(0,i-1);
                }
            }
            
        }
        return prefix;
    }
}
