class Solution {
    public String reverseWords(String s) {
        String words[]=s.trim().split("\\s+");
        String ans="";
        int len=words.length;
        for(int i=len-1;i>=0;i--){
            if(i>0){
                ans=ans+words[i]+" ";
                }
            else{
                ans=ans+words[i];
            }
        }
         return ans;
    }
   
}
