class Solution {
    public String removeOuterParentheses(String s) {
        int j=0;
        int balance=0;
        String str="";
        for(int i=0;i<s.length();i++){
            if(s.charAt(i)=='('){
                balance++;
            }
            else{
                balance--;
            }
            if(balance==0){
                str=str+s.substring(j+1,i);
                j=i+1;
                balance=0;
            }
        }
        return str;
    } 
}
