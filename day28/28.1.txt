class myStack {
    Queue <Integer> q = new LinkedList<>();
    
    void push(int x) {
        int s=q.size();
        q.offer(x);
        for(int i=1;i<=s;i++){
            q.offer(q.peek());
            q.poll();
        }
    }

    void pop() {
       if(!(q.size()==0)){
           q.poll();
       }
    }

    int top() {
       if(!(q.size()==0)){
           return q.peek();
       }
       return -1;
    }

    int size() {
        return q.size();
    }
}
