class Solution {
    public int mySqrt(int x) {
       int ans=0;
       long low=0;
       long high=x;
       long mid=0;
       while(low<=high){
            mid=low +(high-low)/2;
            if(mid*mid==x){
                return (int)mid;
            }
            else if(mid*mid>x){
                high=mid-1;
            }
            else{
                ans=(int)mid;
                low=mid+1;
            }
       }
       return ans;
    }
}
