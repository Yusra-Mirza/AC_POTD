class Solution {
    public int[] countBits(int n) {
        int ans[]= new int[n+1];
        for(int i=0;i<=n;i++){
            ans[i]=noOfSetBits(i);
        }
        return ans;
    }
    public int noOfSetBits(int n){
        int count=0;
        
        while(n>=1){
            count+=(n&1);
            n=n>>1;
        }
        return count;
    }
}
