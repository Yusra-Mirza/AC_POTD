class Solution {
    public String reverse(String S) {
        Stack <Character> st= new Stack<>();
        for(int i=0;i<S.length();i++){
            char ch=S.charAt(i);
            st.push(ch);
        }
        String res="";
        for(int j=st.size()-1;j>=0;j--){
            char ch=st.pop();
            res=res+ch;
        }
        return res;
    }
}
