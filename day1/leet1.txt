class Solution {
    public int numIdenticalPairs(int[] nums) {
        int len=nums.length;
        int good=0;
        for(int i=0;i<len-1;i++){
            int num=nums[i];
            for(int j=i+1;j<len;j++){
                if(nums[j]==num){
                    good++;
                }
            }
        }
        return good;
    }
}
